@login_required def test_detail(request, test_id): test = get_object_or_404(Test, id=test_id) questions = test.questions.all() if request.method == 'POST': score = 0 for question in questions: user_answer = request.POST.get(f'question_{question.id}') if user_answer == question.correct: score += question.points total = sum(q.points for q in questions) percentage = (score / total) * 100 passed = percentage >= test.passing_score # Нәтижені сақтау TestResult.objects.create( user=request.user, test=test, score=score, total=total, percentage=percentage, passed=passed ) # Егер тапсырған болса, сертификат жасау (егер жоқ болса) if passed: existing_cert = Certificate.objects.filter(user=request.user, test=test).first() if not existing_cert: Certificate.objects.create( user=request.user, test=test, score=score, total=total, percentage=percentage ) return render(request, 'test_result.html', { 'test': test, 'score': score, 'total': total, 'percentage': percentage, 'passed': passed }) return render(request, 'test_detail.html', { 'test': test, 'questions': questions })
